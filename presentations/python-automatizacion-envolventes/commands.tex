\newcommand{\htree}[1]{
	\begin{forest}
	for tree={
	fit=band,
	grow'=0,
	parent anchor=children,
	child anchor=parent,
	anchor=parent,
	if n children=0{folder}{},
	edge path'={(!u.parent anchor) -- ++(5pt,0) |- (.child anchor)},
	},
	where n=1{
	calign with current edge
	}{},
	#1
	\end{forest}
}
